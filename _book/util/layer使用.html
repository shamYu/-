移动执法3.0开发文档

layerTool

所有的弹层操作，都在这里封装， 基于 layer.js;

alert 提示

Layer.showAlert('啦啦啦'); //普通的alert 框

Layer,showSucAlert('成功啦', function(){
    alert('点击了确定');
});

Layer.showFailAlert('失败了!', 3000); //3s 后自动关闭

confirm

Layer.showConfirm('确认信息', function(){
    Layer.closeLayer(index);
    alert('点击了确定');
}, function(index){
    Layer.closeLayer(index);
    alert('点击了取消');
})

msg

layer.showMsg('啦啦啦'); //无icon 的提示

//各种的状态的提示信息
layer.showMsg('啦啦啦', 'info');
layer.showMsg('啦啦啦', 'success');
layer.showMsg('啦啦啦', 'fail');

//成功提示
Layer.showSucMsg('啦啦啦', function(){
    alert('提示信息消失了')
})

openDialog

弹出另一个页面

Layer.openDiaLog(title,url,width,height,callback)
  • title 对话框标题(必须)
  • url 对话框URL(必须)
  • width 对话框宽度 默认：800px
  • height 对话框高低 默认：600px
  • callback 窗口销毁时的回调方法

setData 和 getData

openDiaLog 打开的是另一个页面， 两个页面之间的通信通过setData 和getData 完成

Layer.setData('test1');

//在另个页面get
Layer.getData() //test1

loading

打开loading框

Layer.loading();

//带遮蔽框的loading
Layer.loading(0.2)；

closeLayer

关闭弹出层, 以上所有的方法，都会返回一个 index, 关闭时需要使用到

var index = Layer.loading();
layer.closeLayer(index);

//关闭dialog
Layer.closeDialog();

//关闭所有
Layer.closeAll();

Layer.singleSelectWry(function(){
    var returnVal = Layer.getData();
    $("#WRYBH").val(returnVal.WRYBH);
    $("#WRYMC").val(returnVal.WRYMC);
